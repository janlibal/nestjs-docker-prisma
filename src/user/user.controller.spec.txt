import { AuthEmailLoginDto } from "src/app/dto/auth.email.login.dto";
import { UserController } from "./user.controller";
import { UsersService } from "./user.service";

describe('MyController', () => {
    let controller: UserController;
    let service: UsersService;
  
    beforeEach(async () => {
        const createUserDto = {
            email: 'chadwickboseman@email.com',
            password: 'pwd123!'
          }
      service = { create(createUserDto: AuthEmailLoginDto) {createUserDto} } as any;
      controller = new UserController(service);
    });
  
    it('should return user object', () => {
        const createUserDto = {
            email: 'chadwickboseman@email.com',
            password: 'pwd123!'
          }
      expect(controller.login(createUserDto)).toMatchObject(createUserDto)
    });
  });
